Why ditch Web Speech API?

Undocumented quotas & no SLA; GCloud + Vosk gives predictable accuracy and resilience.

3. Parsing logic (Edge / Cloud Function)

# pseudo FastAPI
@post("/parse-slip-line")
def parse(transcript: str):
    tokens = normalise(transcript)            # Hindi→Latin numerals, trim fillers
    qty      = extract_number(tokens[0])
    unit     = extract_unit(tokens[1], default="kg")
    price    = extract_price(tokens[-1])
    product  = fuzzy_match(" ".join(tokens[2:-1]), PRODUCTS)
    return {
        "product_id": product.id,
        "qty": qty,
        "unit": unit,
        "rate": price,
        "total": qty * price
    }

Fuzzy match: Postgres pg_trgm on products(name, aliases) (≈ 100 rows—instant).

4. Data model (PostgreSQL / Supabase-ready)

products      (id PK, name, default_unit, aliases[])
slips         (id UUID PK, slip_no, slip_date, shop_id, customer_name,
               gst_required, subtotal, discount, total)
slip_items    (id PK, slip_id FK, product_id FK, qty, unit, rate, line_total)
price_board   (product_id FK, price, effective_date, PRIMARY KEY(product_id,effective_date))

Indexes: GIN (aliases gin_trgm_ops), BTREE (slip_date), BTREE (shop_id, slip_date).

2.1 Push schema

Create supabase/migrations/0001_init.sql:

create extension if not exists pg_trgm;

create table products (
  id int primary key,
  name text,
  default_unit text,
  aliases text[]
);

create table slips (
  id uuid primary key default gen_random_uuid(),
  slip_no int generated always as identity,
  slip_date date default current_date,
  shop_id int,
  customer_name text,
  gst_required boolean default false,
  subtotal numeric,
  discount numeric,
  total numeric
);

create table slip_items (
  id serial primary key,
  slip_id uuid references slips(id) on delete cascade,
  product_id int references products(id),
  qty numeric,
  unit text,
  rate numeric,
  line_total numeric
);

create table price_board (
  product_id int references products(id),
  price numeric,
  effective_date date default current_date,
  primary key (product_id, effective_date)
);

create index on products using gin(aliases gin_trgm_ops);
create index on slips (slip_date);
create index on slips (shop_id, slip_date);

supabase db push         # runs the migration locally

6. Edge Function: /stt

supabase/functions/stt/index.ts

import { serve } from 'std/server'
import { SpeechClient } from '@google-cloud/speech'
import { decode } from 'base64-arraybuffer'
import { createClient } from '@supabase/supabase-js'

const client = new SpeechClient({
  projectId: Deno.env.get('GCLOUD_PROJECT'),
  credentials: JSON.parse(
    atob(Deno.env.get('GCLOUD_STT_KEY_JSON') ?? '')
  ),
})
const supa = createClient(Deno.env.get('SUPABASE_URL')!, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!)

serve(async req => {
  const { audioBase64, slipId } = await req.json()

  // STREAM to STT
  const [response] = await client.recognize({
    model: 'chirp',                       // high-accuracy model :contentReference[oaicite:3]{index=3}
    languageCodes: ['hi-IN'],
    speechContexts: [{ phrases: ['बासमती', 'अरहर दाल', 'सोना मसूरी'] }],
    config: { audioChannelCount: 1, sampleRateHertz: 8000 },
    content: audioBase64,
  })

  const transcript = response.results?.[0]?.alternatives?.[0]?.transcript ?? ''

  // call parser (could inline here)
  const parseResp = await fetch(`${Deno.env.get('SUPABASE_URL')}/functions/v1/parse-slip-line`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'Authorization': req.headers.get('Authorization')! },
    body: JSON.stringify({ transcript, slipId }),
  })
  return new Response(await parseResp.text(), { headers: { 'Content-Type': 'application/json' } })
})
